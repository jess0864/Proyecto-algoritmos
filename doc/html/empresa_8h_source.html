353  static void merge(vector<Empresa*>& arr, int left, int mid, int right) {
354  int n1 = mid - left + 1;
355  int n2 = right - mid;
356  vector<Empresa*> L(n1), R(n2);
357  for (int i = 0; i < n1; i++) L[i] = arr[left + i];
358  for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];
359  int i = 0, j = 0, k = left;
360  while (i < n1 && j < n2) {
361  if (L[i]->precioActual > R[j]->precioActual) arr[k++] = L[i++];
362  else arr[k++] = R[j++];
363  }
364  while (i < n1) arr[k++] = L[i++];
365  while (j < n2) arr[k++] = R[j++];
366  }
