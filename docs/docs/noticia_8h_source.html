268 string generarFecha(int diasDesdeInicio = 0) {
269  int year = 2025;
270  int month = 5;
271  int day = 1 + diasDesdeInicio;
272 
273  // Ajustar el día y el mes si se pasa del final del mes
274  while (true) {
275  int diasMes;
276  if (month == 2) diasMes = 28;
277  else if (month == 4 || month == 6 || month == 9 || month == 11) diasMes = 30;
278  else diasMes = 31;
279  if (day > diasMes) {
280  day -= diasMes;
281  month++;
282  if (month > 12) {
283  month = 1;
284  year++;
285  }
286  } else {
287  break;
288  }
289  }
290 
291  string fecha = "";
292  fecha += to_string(year) + "-";
293  if (month < 10) fecha += "0";
294  fecha += to_string(month) + "-";
295  if (day < 10) fecha += "0";
296  fecha += to_string(day);
297  return fecha;
298 }
