346  static void merge(vector<Empresa*>& arr, int left, int mid, int right) {
347  int n1 = mid - left + 1;
348  int n2 = right - mid;
349  vector<Empresa*> L(n1), R(n2);
350  for (int i = 0; i < n1; i++) L[i] = arr[left + i];
351  for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];
352  int i = 0, j = 0, k = left;
353  while (i < n1 && j < n2) {
354  if (L[i]->precioActual > R[j]->precioActual) arr[k++] = L[i++];
355  else arr[k++] = R[j++];
356  }
357  while (i < n1) arr[k++] = L[i++];
358  while (j < n2) arr[k++] = R[j++];
359  }

367  static void mergeSort(vector<Empresa*>& arr, int left, int right) {
368  if (left < right) {
369  int mid = left + (right - left) / 2;
370  mergeSort(arr, left, mid);
371  mergeSort(arr, mid + 1, right);
372  merge(arr, left, mid, right);
373  }
